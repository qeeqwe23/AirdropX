    <!-- 设计说明：
         1. 采用 turbine_engine + direct thruster 架构，推力由发动机模型直接计算；
         2. 配置一个小燃油箱，影响总重很小，但可避免无油导致发动机熄火；
         3. 发动机安装在气动参考点，方向与机身轴线一致，保证推力沿纵向。 -->

    <!-- [4] 飞行控制系统 (Flight Control System) -->
    <!-- 核心作用：定义无人机的控制通道，实现飞控指令到执行机构的映射；
         此处仅实现纵向控制的两个核心通道：俯仰通道（升降舵）和油门通道，满足级联控制需求。 -->
    <flight_control name="HY100_FCS">
        <!-- 俯仰控制通道 (Pitch)：处理升降舵指令，输出升降舵实际偏角 -->
        <channel name="Pitch">
            <!-- 信号求和器：对升降舵指令进行限幅 -->
            <summer name="Elevator Sum">
                <input> fcs/elevator-cmd-norm </input>  <!-- 输入：归一化升降舵指令（来自飞控算法） -->
                <clipto> <min> -1.0 </min> <max> 1.0 </max> </clipto>  <!-- 限幅范围：[-1, 1] -->
            </summer>
            <!-- 舵面缩放器：将归一化指令转换为实际的舵面偏角（度） -->
            <aerosurface_scale name="Elevator Control">
                <input>fcs/elevator-cmd-norm</input>  <!-- 输入：归一化升降舵指令 -->
                <range>                               <!-- 输出范围：升降舵的最大偏角 -->
                    <min> -30 </min>  <!-- 最小偏角：-30 度（上仰） -->
                    <max>  30 </max>  <!-- 最大偏角：+30 度（下俯） -->
                </range>
                <output>fcs/elevator-pos-deg</output>  <!-- 输出：升降舵实际偏角（度），传递给气动力模型 -->
            </aerosurface_scale>
        </channel>

        <!-- 油门控制通道 (Throttle)：最小配置，实现指令的直通传递 -->
        <!-- 设计说明：JSBSim 的发动机通常读取 throttle-pos-norm 作为节流阀位置；
             此处将 throttle-cmd-norm 直通映射到 throttle-pos-norm，
             Python 侧飞控算法可直接写 throttle-cmd-norm，实现油门的闭环控制。 -->
        <channel name="Throttle">
            <pure_gain name="Throttle PassThrough">
                <input>fcs/throttle-cmd-norm</input>  <!-- 输入：归一化油门指令（来自飞控算法） -->
                <gain>1</gain>                        <!-- 增益：1，直通传递 -->
                <output>fcs/throttle-pos-norm</output> <!-- 输出：归一化油门位置，传递给发动机模型 -->
            </pure_gain>
        </channel>
    </flight_control>

    <!-- [5] 空气动力学模型 (Aerodynamics) -->
    <!-- 核心作用：定义无人机的气动力和力矩计算逻辑，是飞行动力学仿真的核心；
         此处仅实现纵向气动力/力矩：升力、阻力、俯仰力矩，满足纵向控制的需求；
         所有气动力的计算都基于动压、机翼面积和相应的系数，力矩还需乘以参考弦长。 -->
    <aerodynamics>
        <!-- 升力轴 (LIFT)：沿机身垂向，向上为正，决定无人机的上升和下降 -->
        <axis name="LIFT">
            <function name="aero/force/Lift_alpha">
                <description>Lift due to alpha</description>  <!-- 升力由迎角 alpha 决定 -->
                <product>  <!-- 升力计算公式：L = q_bar * S_w * C_L(alpha) -->
                    <property>aero/qbar-psf</property>    <!-- 动压：单位为磅/平方英尺 -->
                    <property>metrics/Sw-sqft</property>  <!-- 机翼面积：单位为平方英尺 -->
                    <!-- 查表函数：迎角 alpha 到升力系数 C_L 的非线性映射 -->
                    <table>
                        <independentVar lookup="row">aero/alpha-rad</independentVar>  <!-- 独立变量：迎角（弧度） -->
                        <tableData>  <!-- 查表数据：[迎角(rad), 升力系数C_L] -->
                            -0.20  -0.750  <!-- 迎角 -0.2 rad，C_L = -0.750（负升力） -->
                             0.00   0.250  <!-- 迎角 0 rad，C_L = 0.250（零升力系数） -->
                             0.26   1.400  <!-- 迎角 0.26 rad，C_L = 1.400（最大升力系数，失速迎角） -->
                             0.60   0.710  <!-- 迎角 0.6 rad，C_L = 0.710（失速后，升力系数下降） -->
                        </tableData>
                    </table>
                </product>
            </function>
        </axis>

        <!-- 阻力轴 (DRAG)：沿机身纵向，向后为正，决定无人机的空速和能量消耗 -->
        <axis name="DRAG">
            <!-- 基本阻力 (Drag_basic)：零升阻力，与升力无关，是飞机的基本阻力 -->
            <function name="aero/force/Drag_basic">
                <description>Drag at zero lift</description>
                <product>  <!-- 计算公式：D_basic = q_bar * S_w * C_D0 -->
                    <property>aero/qbar-psf</property>
                    <property>metrics/Sw-sqft</property>
                    <value> 0.025 </value>  <!-- 零升阻力系数 C_D0 = 0.025 -->
                </product>
            </function>
            <!-- 诱导阻力 (Drag_induced)：由升力产生的阻力，与升力系数的平方成正比 -->
            <function name="aero/force/Drag_induced">
                <description>Induced drag</description>
                <product>  <!-- 计算公式：D_induced = q_bar * S_w * C_L² * K -->
                    <property>aero/qbar-psf</property>
                    <property>metrics/Sw-sqft</property>
                    <property>aero/cl-squared</property>  <!-- 升力系数的平方 C_L² -->
                    <value> 0.043 </value>  <!-- 诱导阻力因子 K = 0.043 -->
                </product>
            </function>
        </axis>

        <!-- 俯仰力矩轴 (PITCH)：绕机身横向轴，决定无人机的俯仰姿态，是纵向控制的核心 -->
        <axis name="PITCH">
            <!-- 迎角俯仰力矩 (Pitch_alpha)：由迎角产生的俯仰力矩，决定飞机的静稳定性 -->
            <function name="aero/moment/Pitch_alpha">
                <description>Pitch moment due to alpha</description>
                <product>  <!-- 计算公式：M_alpha = q_bar * S_w * c_bar * C_m_alpha * alpha -->
                    <property>aero/qbar-psf</property>
                    <property>metrics/Sw-sqft</property>
                    <property>metrics/cbarw-ft</property>  <!-- 机翼平均气动弦长，单位为英尺 -->
                    <property>aero/alpha-rad</property>    <!-- 迎角，单位为弧度 -->
                    <value> -0.5 </value>  <!-- 力矩系数 C_m_alpha = -0.5，负号表示静稳定 -->
                </product>
                <!-- 静稳定说明：迎角增大时，产生低头力矩，使飞机自动恢复到平衡迎角 -->
            </function>
            <!-- 升降舵俯仰力矩 (Pitch_elevator)：由升降舵偏角产生的俯仰力矩，是控制飞机俯仰的核心 -->
            <function name="aero/moment/Pitch_elevator">
                <description>Pitch moment due to elevator</description>
                <product>  <!-- 计算公式：M_elevator = q_bar * S_w * c_bar * C_m_elevator * delta_elevator -->
                    <property>aero/qbar-psf</property>
                    <property>metrics/Sw-sqft</property>
                    <property>metrics/cbarw-ft</property>
                    <property>fcs/elevator-pos-rad</property>  <!-- 升降舵偏角，单位为弧度 -->
                    <value> -1.2 </value>  <!-- 力矩系数 C_m_elevator = -1.2 -->
                </product>
                <!-- 控制逻辑说明：升降舵上偏（delta_elevator 为负）时，产生抬头力矩（M_elevator 为正）；
                     升降舵下偏（delta_elevator 为正）时，产生低头力矩（M_elevator 为负），符合常规飞机控制逻辑。 -->
            </function>
            <!-- 俯仰阻尼力矩 (Pitch_damp)：由俯仰角速度产生的阻尼力矩，抑制飞机的俯仰振荡，提高稳定性 -->
            <function name="aero/moment/Pitch_damp">
                <description>Pitch moment due to pitch rate</description>
                <product>  <!-- 计算公式：M_damp = q_bar * S_w * c_bar * (c_bar / 2V) * C_m_q * q -->
                    <property>aero/qbar-psf</property>
                    <property>metrics/Sw-sqft</property>
                    <property>metrics/cbarw-ft</property>
                    <property>aero/ci2vel</property>  <!-- c_bar / (2 * V)，速度相关的阻尼因子 -->
                    <property>velocities/q-aero-rad_sec</property>  <!-- 俯仰角速度，单位为弧度/秒 -->
                    <value> -15 </value>  <!-- 阻尼力矩系数 C_m_q = -15，负号表示阻尼作用 -->
                </product>
                <!-- 阻尼作用说明：俯仰角速度越大，阻尼力矩越大，使飞机的俯仰振荡快速衰减，提高飞行稳定性。 -->
            </function>
        </axis>
    </aerodynamics>

</fdm_config>
